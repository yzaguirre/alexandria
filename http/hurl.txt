Install
https://hurl.dev/docs/installation.html
	wget https://github.com/Orange-OpenSource/hurl/releases/download/1.8.0/hurl_1.8.0_amd64.deb
	dpkg -i hurl_1.8.0_amd64.deb
Windows
	winget install Orange-OpenSource.Hurl
Installer
	wget https://github.com/Orange-OpenSource/hurl/releases/download/1.8.0/hurl-1.8.0-win64-installer.exe
	start hurl-1.8.0-win64-installer.exe
Homebrew
	brew install hurl

Usage
	hurl --progress --summary --html . test1.hurl

Export hurl file
	hurlfmt test.hurl --format json | jq
	hurlfmt access_file_list.hurl --format json -o access_file_list_POSTMAN.json
