Install
	wget https://github.com/caddyserver/caddy/releases/download/v2.4.6/caddy_2.4.6_linux_amd64.deb
	wget https://github.com/caddyserver/caddy/releases/download/v2.4.6/caddy_2.4.6_checksums.txt
	sha512sum -c --ignore-missing caddy_2.4.6_checksums.txt
	dpkg -i caddy_2.4.6_linux_amd64.deb
	systemctl disable caddy
	systemctl stop caddy

Windows

MacOS
	brew install caddy
To have launchd start caddy now and restart at login:
	brew services start caddy
Or, if you don't want/need a background service you can just run:
	caddy run --config /usr/local/etc/Caddyfile

Reverse proxy for IIS express
	vim Caddyfile
http://192.168.0.5:8080 {
	reverse_proxy localhost:62555 {
		header_up Host localhost:62555
	}
	log {
		output file .\proxylog.log
		format console
		level INFO
	}
}
Run
	caddy.exe run --config Caddyfile
