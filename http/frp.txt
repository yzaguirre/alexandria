Socks5 behind a http proxy. npm can use it.
Set up server component
	vim frps.ini
[common]
bind_port = 7000

	frps -c frps.ini

Set up client component
	vim frpc.ini
[common]
server_addr = 127.0.0.1
server_port = 7000

[http_proxy]
type = tcp
remote_port = 8080
plugin = http_proxy

	frpc -c frpc.ini
