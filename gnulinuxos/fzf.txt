	apt install fzf
Download binary
	wget https://github.com/junegunn/fzf/releases/download/0.35.0/fzf-0.35.0-linux_amd64.tar.gz
	wget https://github.com/junegunn/fzf/releases/download/0.35.0/fzf_0.35.0_checksums.txt
	sha256sum -c --ignore-missing fzf_0.35.0_checksums.txt
	tar xf fzf-0.35.0-linux_amd64.tar.gz -C /usr/local/bin
Windows
	choco install fzf
Homebrew
	brew install fzf
To install useful key bindings and fuzzy completion:
	$(brew --prefix)/opt/fzf/install
