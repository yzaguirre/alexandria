You need to check if your script is currently running in the system -
you can do it with ps
#!/bin/bash
cmd=namedScript
runningProcs=`ps --no-headers -C${cmd}`
count=`echo $runningProcs|wc -l`
if [ $count -gt 1 ]; then
	echo "Previous $cmd is still running."
	exit 1
fi
