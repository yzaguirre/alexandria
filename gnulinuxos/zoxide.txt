Install Bash
eval "$(zoxide init bash)"

Install Fish
	vim ~/.config/fish/config.fish
zoxide init fish | source

Install Powershell
Invoke-Expression (& {
    $hook = if ($PSVersionTable.PSVersion.Major -lt 6) { 'prompt' } else { 'pwd' }
    (zoxide init --hook $hook powershell | Out-String)
})
