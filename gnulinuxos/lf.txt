Termux
	apt install lf
Windows
	choco install lf
	scoop install lf
Homebrew
	brew install lf

Configuration
Example file
	mkdir ~/.config/lf
	wget -O ~/.config/lf/lfrc https://github.com/gokcehan/lf/raw/master/etc/lfrc.example
	vim ~/.config/lf/lfrc

	cd lf-gadgets/lf-ueberzug
	cp lf-* /usr/local/bin
	mkdir ~/.config/lf-ueberzug
	cp lfrc-ueberzug ~/.config/lf-ueberzug
New alias
	vim ~/.bash_aliases
