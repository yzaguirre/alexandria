ES6 -> ES2015
	npm init
or	echo {} > package.json
	npm install --save-dev @babel/core @babel/cli
Add build script
"scripts": {
	"build": "babel src -d lib"
}
Edit source file
	mkdir src && cd src
	touch Person.js
Build
	npm run build

	vim .babelrc

  "presets": ["@babel/preset-env"]
}
	npm run build
