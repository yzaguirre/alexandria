Configuration
Linux
${XDG_CONFIG_HOME:-${HOME}/.config}/streamlink/config
MacOS
${HOME}/Library/Application Support/streamlink/config
Windows
%APPDATA%\streamlink\config

# Player options
player=mpv --cache 2048
