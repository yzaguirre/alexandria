Verify
	gpg --with-fingerprint KEY
	gpg --import KEY
	gpg --verify SHA256SUMS.asc
	gpg --verify SHA512SUMS.asc
	sha256sum thunderbird-102.6.1.tar.bz2
	sha512sum thunderbird-102.6.1.tar.bz2
	grep "linux-x86_64/es-ES/thunderbird-102.6.1" MD5SUMS
	grep "linux-x86_64/es-ES/thunderbird-102.6.1" SHA1SUMS
	grep "linux-x86_64/es-ES/thunderbird-102.6.1" SHA512SUMS
	tar xf thunderbird-102.6.1.tar.bz2 -C /VMs/opt
	ln -s /VMs/opt/thunderbird/thunderbird-bin /usr/local/bin/thunderbird
