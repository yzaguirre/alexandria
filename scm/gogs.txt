http://gogs.io/docs/installation
	wget https://github.com/gogits/gogs/releases/download/v0.6.1/linux_amd64.zip
	mkdir -p /home/git/.ssh
	unzip linux_amd64.zip -d /opt
Create User to run gogs
	useradd -m -d /home/git --uid 1005 -s /bin/bash git
x	usermod -s /usr/bin/git-shell git
x	usermod -s /bin/bash git
	chown git:git -R /home/git
	chown git:git -R /opt/gogs
Run
	su git
	cd /opt/gogs
	./gogs web
	
http://gogs.io/docs/advanced/configuration_cheat_sheet.html
Configuration file is located in <install folder>/custom/conf/app.ini
Repository location
Database location
Public key to enable commit over ssh
	cd /opt/gogs
	mkdir -p custom/conf
	cd custom/conf
	vim app.ini
[repository]
ROOT = /var/gogs/git
[database]
PATH = /var/gogs/git/gogs.db
	su git
	/var/gogs/bin/gogs web
