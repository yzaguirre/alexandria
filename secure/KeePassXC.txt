	apt install qttools5-dev g++ cmake libgcrypt11-dev libsodium-dev libargon2-0-dev libcurl4-openssl-dev libqt5x11extras5-dev libxi-dev libxtst-dev qtbase5-dev qttools5-dev-tools libgcrypt20-dev
	git clone https://github.com/keepassxreboot/keepassxc.git
	cd keepassxc
	git checkout 2.5.4
	mkdir build
	cd build
	cmake -DWITH_TESTS=OFF -DCMAKE_INSTALL_PREFIX=/usr/local -DCMAKE_BUILD_TYPE=Release -DWITH_XC_HTTP=ON -DWITH_TESTS=OFF ..
	make -j 2
	sudo make install
Install in /usr/local/share/keepassxc/

	wget https://github.com/keepassxreboot/keepassxc/releases/download/2.6.0/KeePassXC-2.6.0-x86_64.AppImage
	wget https://github.com/keepassxreboot/keepassxc/releases/download/2.6.0/KeePassXC-2.6.0-x86_64.AppImage.DIGEST
	wget https://github.com/keepassxreboot/keepassxc/releases/download/2.6.0/KeePassXC-2.6.0-x86_64.AppImage.sig
	gpg --verify KeePassXC-2.6.0-x86_64.AppImage.sig KeePassXC-2.6.0-x86_64.AppImage
	sha256sum -c KeePassXC-2.6.0-x86_64.AppImage.DIGEST
	chmod u+x KeePassXC-2.6.0-x86_64.AppImage
	mv KeePassXC-2.6.0-x86_64.AppImage ~/appimage
	ln -s ~/appimage/KeePassXC-2.6.0-x86_64.AppImage /usr/local/bin/keepassxc
