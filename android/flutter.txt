	wget https://storage.googleapis.com/flutter_infra/releases/stable/linux/flutter_linux_v1.2.1-stable.tar.xz
	tar xf flutter_linux_v1.2.1-stable.tar.xz -C /opt
Dart is at /opt/flutter/bin/cache/dart-sdk
Edit PATH
export PATH=$PATH:/opt/flutter/bin

Run flutter
	flutter config --no-analytics
	flutter config
	flutter doctor -v
	flutter doctor --android-licenses
	flutter devices
Test app
	flutter create myapp
	cd myapp
	flutter devices
	flutter run
Format a file
	flutter format <filename>
