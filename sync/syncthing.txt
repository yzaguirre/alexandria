Install amd64
	tar xf syncthing-linux-amd64-v1.13.1.tar.gz -C /opt
	mv /opt/{syncthing-linux-amd64-v1.13.1,syncthing}
	ln -s /opt/syncthing/syncthing /usr/local/bin/syncthing
Install raspberry pi
	mkdir ~/.syncthing
	tar xf syncthing-linux-arm-v1.13.1.tar.gz -C ~/
	mv ~/syncthing-linux-arm-v1.13.1 .syncthing
	ln -s ~/.syncthing/syncthing /usr/local/bin/syncthing
Debian repo
http://apt.syncthing.net/
	curl -s https://syncthing.net/release-key.txt | apt-key add -
	echo "deb http://apt.syncthing.net/ syncthing release" >> /etc/apt/sources.list.d/syncthing.list
	apt update
	apt install syncthing

Install MacOS
	unzip syncthing-macos-universal-v1.13.1.zip
	cd syncthing-macos-universal-v1.13.1
	ruplacer "/USERNAME" "/david" etc/macos-launchd/syncthing.plist
	mv etc/macos-launchd/syncthing.plist ~/Library/LaunchAgents/syncthing.plist
	mv syncthing ~/bin
Stop syncthing
https://forum.syncthing.net/t/keeping-syncthing-running-mac-os-x-launchdaemon/407/29
	launchctl unload ~/Library/LaunchAgents/syncthing.plist
Start syncthing
	launchctl load ~/Library/LaunchAgents/syncthing.plist
