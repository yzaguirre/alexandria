Binary

	wget https://download.blender.org/release/Blender2.93/blender-2.93.1.md5
	wget https://download.blender.org/release/Blender2.93/blender-2.93.1.sha256
	md5sum -c --ignore-missing blender-2.93.1.md5
	sha256sum -c --ignore-missing blender-2.93.1.sha256
	tar xf blender-2.93.1-linux-x64.tar.xz -C /opt
	mv /opt/blender{-2.93.1-linux-x64,}
	ln -s /opt/blender/blender /usr/local/bin
