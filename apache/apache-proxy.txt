Alias /redmine "/usr/local/share/redmine-2.4.2" 
<Proxy *>
  RequestHeader set X_FORWARDED_PROTO 'https'
  Order deny,allow
  Allow from all
</Proxy>

ProxyRequests Off

ProxyPass /redmine/ http://127.0.0.1:3000/redmine/
ProxyPassReverse /redmine http://127.0.0.1:3000/redmine/
