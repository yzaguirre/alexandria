#!/bin/bash

useradd -d /var/www/pedro -s /bin/false pedro
passwd juan
passwd paco
passwd luis
passwd pedro
mkdir -p /var/www/{juan,paco,luis,pedro}/doc
rmdir /var/www/{juan,paco,luis,pedro}/doc
rmdir /var/www/{juan,paco,luis,pedro}
