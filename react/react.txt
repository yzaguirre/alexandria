	yarn add typescript

	yarn add --dev @babel/preset-typescript
Sass
	yarn add --dev sass style-loader css-loader sass-loader
Saas file in src/styles/main.scss
Rule
	{
		test: /\.scss$/,
		exclude: /node_modules/,
		use: ['style-loader', 'css-loader', 'sass-loader']
	}
postcss
	yarn add --dev postcss style-loader css-loader postcss-loader
Rule
	{
		test: /\.css$/,
		exclude: /node_modules/,
		use: ['style-loader', 'css-loader']
	}
