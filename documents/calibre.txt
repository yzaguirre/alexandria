Disable calibre as default PDF viewer
	mv /usr/share/applications/ebook-viewer.desktop{,.disabled}
	mv /usr/local/share/applications/ebook-viewer.desktop{,.disabled}

Binary install
https://calibre-ebook.com/download_linux
	sudo su
	wget https://download.calibre-ebook.com/linux-installer.py
	cat linux-installer.py | sudo python -c "import sys; main=lambda:sys.stderr.write('Download failed\n'); exec(sys.stdin.read()); main()"

	/usr/bin/calibre-uninstall
